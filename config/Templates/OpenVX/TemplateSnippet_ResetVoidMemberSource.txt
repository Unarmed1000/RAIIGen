    void ##CLASS_NAME##::Reset(const ResetMode resetMode)
    {
      // Free any currently allocated resource
      if (IsValid())
        Reset();

      if( resetMode == ResetMode::Destroy )
        return;

      // Since we want to ensure that the resource is left untouched on error we use a local variable as a intermediary
      const ##RESOURCE_TYPE## ##RESOURCE_INTERMEDIARY_NAME## = ##CREATE_FUNCTION##(##CREATE_FUNCTION_ARGUMENTS##);
      if( ##RESOURCE_INTERMEDIARY_NAME## == ##DEFAULT_VALUE## )
        throw std::runtime_error("##CREATE_FUNCTION## failed");

      // Everything is ready, so assign the members##RESET_SET_MEMBERS##
    }
